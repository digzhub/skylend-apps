Place the following script at ~/.termux/boot/start-skylend.sh and make it executable:

#!/data/data/com.termux/files/usr/bin/sh
termux-wake-lock
cd /data/data/com.termux/files/home/skylend
if command -v http-server >/dev/null 2>&1; then
  nohup http-server -p 8080 > /sdcard/skylend_server.log 2>&1 &
else
  nohup python3 -m http.server 8080 > /sdcard/skylend_server.log 2>&1 &
fi
